```
import os
import pandas as pd
from pathlib import Path

def create_dataset_index(raw_data_dir, output_csv):
    """Сканує директорію з даними та створює CSV з шляхами до NIfTI файлів."""
    data = []
    base_path = Path(raw_data_dir)
    
    # Ітеруємося по папках пацієнтів
    for patient_dir in base_path.iterdir():
        if patient_dir.is_dir():
            patient_id = patient_dir.name
            
            # Формуємо шляхи до кожної модальності та маски
            paths = {
                'patient_id': patient_id,
                't1': str(patient_dir / f"{patient_id}-t1n.nii.gz"), # Перевірте точні суфікси файлів у вашому архіві
                't1c': str(patient_dir / f"{patient_id}-t1c.nii.gz"),
                't2': str(patient_dir / f"{patient_id}-t2w.nii.gz"),
                'flair': str(patient_dir / f"{patient_id}-t2f.nii.gz"),
                'seg': str(patient_dir / f"{patient_id}-seg.nii.gz")
            }
            
            # Перевіряємо, чи існують всі необхідні файли
            if all(os.path.exists(p) for p in list(paths.values())[1:]):
                data.append(paths)
                
    df = pd.DataFrame(data)
    df.to_csv(output_csv, index=False)
    print(f"Індекс створено! Знайдено {len(df)} пацієнтів. Збережено у {output_csv}")

if __name__ == "__main__":
```

```
import os
import pandas as pd
from pathlib import Path

def create_dataset_index(raw_data_dir, output_csv):
    data = []
    base_path = Path(raw_data_dir)
    
    # Шукаємо всі файли масок
    for seg_path in base_path.rglob("*-seg.nii.gz"):
        patient_dir = seg_path.parent

        prefix = seg_path.name.replace('-seg.nii.gz', '')
        
        # Дивимося, які файли реально є в папці
        files_in_dir = [f.name for f in patient_dir.iterdir() if f.is_file()]
        
        # Знайдемо точні імена файлів модальностей
        t1_file = next((f for f in files_in_dir if f.endswith('t1n.nii.gz')), None)
        t1c_file = next((f for f in files_in_dir if f.endswith('t1c.nii.gz')), None)
        t2_file = next((f for f in files_in_dir if f.endswith('t2w.nii.gz')), None)
        flair_file = next((f for f in files_in_dir if f.endswith('t2f.nii.gz')), None)

        if t1_file and t1c_file and t2_file and flair_file:
            paths = {
                'patient_id': prefix,
                't1': str(patient_dir / t1_file), 
                't1c': str(patient_dir / t1c_file),
                't2': str(patient_dir / t2_file),
                'flair': str(patient_dir / flair_file),
                'seg': str(seg_path)
            }
            data.append(paths)
            
    df = pd.DataFrame(data)
    os.makedirs(os.path.dirname(output_csv), exist_ok=True)
    df.to_csv(output_csv, index=False)
    print(f"Індекс створено! Знайдено {len(df)} пацієнтів з повним набором даних. Збережено у {output_csv}")
```

Те, що маска праворуч повністю чорна, має дуже просте і логічне медичне пояснення:

МРТ — це 3D-об'єм. Ми взяли навмання 75-й зріз по осі Z (по висоті голови). Але пухлина не займає весь мозок. Вона може знаходитися вище (наприклад, на 100-му зрізі) або нижче (на 40-му). На 75-му зрізі у цього конкретного пацієнта просто здорові тканини, тому лікарі не виділили там жодного пікселя пухлини. Відповідно, маска складається виключно з нулів (чорний колір).

Щоб побачити пухлину, нам потрібно трохи "прокачати" наш код. Замість того, щоб брати 75-й зріз наосліп, давайте навчимо скрипт **автоматично знаходити той зріз, де пухлина найбільша**.

Ось як треба змінити шматок коду в `explore_data.py`. Замініть рядок `slice_idx = 75` на цей блок:

```
    # Шукаємо найкращий зріз автоматично!
    # Рахуємо кількість пікселів пухлини (де значення > 0) для кожного зрізу (вісь Z - це індекс 2)
    tumor_pixels_per_slice = (seg_data > 0).sum(axis=(0, 1))

    best_slice_idx = tumor_pixels_per_slice.argmax()
```

Сирі МРТ-знімки мають дві проблеми для нейромереж:

1. **Забагато порожнечі:** Голова займає лише частину 3D-масиву, решта — це чорні пікселі (повітря навколо голови). Вони лише витрачають дорогоцінну оперативну пам'ять вашого комп'ютера. Ми їх відріжемо (**Cropping**).
    
2. **Різна яскравість:** МРТ-сканери не мають стандартних одиниць вимірювання яскравості (як, наприклад, шкала Гаунсфілда у КТ). У одного пацієнта пікселі мозку можуть мати значення від 0 до 500, у іншого — від 0 до 2000. Ми зведемо їх до єдиного масштабу за допомогою **Z-нормалізації**, щоб наша нейромережа не "заплуталася".

```
import numpy as np

def get_bounding_box(image_data):
    """
    Шукає межі мозку у 3D-просторі, ігноруючи чорний фон (пікселі <= 0).
    Повертає мінімальні та максимальні координати для обрізки.
    """
    # Знаходимо індекси всіх пікселів, які є частиною мозку
    coords = np.array(np.nonzero(image_data > 0))
    
    # Знаходимо крайні точки по осях (X, Y, Z)
    min_coords = coords.min(axis=1)
    max_coords = coords.max(axis=1) + 1 # +1 для правильного зрізу в Python
    
    return min_coords, max_coords

def crop_volume(image_data, min_coords, max_coords):
    """Обрізає 3D-масив за заданими координатами."""
    return image_data[
        min_coords[0]:max_coords[0],
        min_coords[1]:max_coords[1],
        min_coords[2]:max_coords[2]
    ]

def normalize_intensity(image_data):
    """
    Z-score нормалізація: (X - Mean) / Std.
    ВАЖЛИВО: Статистика рахується ТІЛЬКИ по пікселях мозку, 
    щоб чорний фон не спотворював середнє значення.
    """
    # Створюємо маску мозку (там, де пікселі яскравіші за 0)
    brain_mask = image_data > 0
    brain_pixels = image_data[brain_mask]
    
    # Рахуємо середнє та стандартне відхилення лише для тканин мозку
    mean = np.mean(brain_pixels)
    std = np.std(brain_pixels)
    
    # Створюємо порожній масив такого ж розміру
    normalized_data = np.zeros_like(image_data, dtype=np.float32)
    
    # Застосовуємо формулу нормалізації тільки до мозку, фон залишається нулем
    if std > 0: # Захист від ділення на нуль
        normalized_data[brain_mask] = (image_data[brain_mask] - mean) / std
        
    return normalized_data
```

```
import torch
from torch.utils.data import Dataset
import nibabel as nib
import pandas as pd
import numpy as np
from src.data.preprocessing import get_bounding_box, crop_volume, normalize_intensity

class BraTSDataset(Dataset):
    def __init__(self, csv_file, patch_size=(64, 64, 64)):
        """
        csv_file: шлях до нашого dataset_index.csv
        patch_size: розмір 3D-кубика, який ми будемо "вирізати" для навчання
        """
        self.data_index = pd.read_csv(csv_file)
        self.patch_size = patch_size

    def __len__(self):
        return len(self.data_index)

    def __getitem__(self, idx):
        patient = self.data_index.iloc[idx]
        
        # 1. Завантаження (float32 для економії оперативки)
        t1 = nib.load(patient['t1']).get_fdata().astype(np.float32)
        t1c = nib.load(patient['t1c']).get_fdata().astype(np.float32)
        t2 = nib.load(patient['t2']).get_fdata().astype(np.float32)
        flair = nib.load(patient['flair']).get_fdata().astype(np.float32)
        seg = nib.load(patient['seg']).get_fdata().astype(np.uint8) # Маска має цілі класи 0,1,2,4
        
        # 2. Знаходимо межі мозку (достатньо по одній модальності, наприклад FLAIR)
        min_c, max_c = get_bounding_box(flair)
        
        # ВАЖЛИВО: Обрізаємо ВСІ модальності та маску за ОДНАКОВИМИ координатами,
        # щоб вони не "роз'їхалися" у просторі
        t1 = crop_volume(t1, min_c, max_c)
        t1c = crop_volume(t1c, min_c, max_c)
        t2 = crop_volume(t2, min_c, max_c)
        flair = crop_volume(flair, min_c, max_c)
        seg = crop_volume(seg, min_c, max_c)
        
        # 3. Нормалізація (маску не нормалізуємо, бо там класи!)
        t1 = normalize_intensity(t1)
        t1c = normalize_intensity(t1c)
        t2 = normalize_intensity(t2)
        flair = normalize_intensity(flair)
        
        # 4. Об'єднання у багатоканальний тензор [4 канали, X, Y, Z]
        image_volume = np.stack([t1, t1c, t2, flair], axis=0)
        
        # 5. Вирізання випадкового 3D-патчу
        ph, pw, pd_size = self.patch_size
        _, h, w, d = image_volume.shape
        
        # Генеруємо випадкові початкові координати для кубика
        start_h = np.random.randint(0, h - ph)
        start_w = np.random.randint(0, w - pw)
        start_d = np.random.randint(0, d - pd_size)
        
        img_patch = image_volume[:, start_h:start_h+ph, start_w:start_w+pw, start_d:start_d+pd_size]
        # Масці додаємо технічний вимір каналу (щоб було [1, X, Y, Z])
        mask_patch = seg[np.newaxis, start_h:start_h+ph, start_w:start_w+pw, start_d:start_d+pd_size]
        
        # 6. Конвертація у тензори PyTorch
        return torch.from_numpy(img_patch), torch.from_numpy(mask_patch)
```

Переходимо до **Кроку 1 (Тиждень 2): Створення 3D U-Net**.

Оскільки наші дані — це 3D-кубики (патчі 64x64x64), звичайна 2D нейромережа нам не підійде. Ми використаємо архітектуру **3D U-Net**, яка є золотим стандартом у медичній сегментації. Вона складається з:

1. **Енкодера (Encoder):** Стискає зображення, виділяючи важливі ознаки (де знаходиться пухлина, яка її текстура).
    
2. **Декодера (Decoder):** Розширює стиснуті дані назад до розміру 64x64x64, створюючи точну маску кожного пікселя.
    

#### Дії у вашому робочому середовищі:

Створіть новий файл `src/models/unet3d.py` і вставте цей базовий каркас моделі. Я налаштував її так, щоб вона була "легкою" і не перевантажила вашу відеокарту.

Python

```
import torch
import torch.nn as nn

class DoubleConv3D(nn.Module):
    """(Conv3D -> BatchNorm -> ReLU) * 2"""
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv3d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm3d(out_channels),
            nn.ReLU(inplace=True),
            nn.Conv3d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm3d(out_channels),
            nn.ReLU(inplace=True)
        )

    def forward(self, x):
        return self.double_conv(x)

class UNet3D(nn.Module):
    def __init__(self, in_channels=4, out_channels=4, features=[16, 32, 64, 128]):
        """
        in_channels=4 (наші 4 модальності: T1, T1ce, T2, FLAIR)
        out_channels=4 (наші класи: 0-Фон, 1-Некроз, 2-Набряк, 4-Активна пухлина)
        features: кількість каналів на кожному рівні (робимо невеликі для економії пам'яті)
        """
        super().__init__()
        self.downs = nn.ModuleList()
        self.ups = nn.ModuleList()
        self.pool = nn.MaxPool3d(kernel_size=2, stride=2)

        # Будуємо Енкодер (Шлях вниз)
        in_c = in_channels
        for feature in features:
            self.downs.append(DoubleConv3D(in_c, feature))
            in_c = feature

        # Ботлнек (Найнижчий рівень мережі)
        self.bottleneck = DoubleConv3D(features[-1], features[-1] * 2)

        # Будуємо Декодер (Шлях вгору)
        for feature in reversed(features):
            self.ups.append(
                nn.ConvTranspose3d(feature * 2, feature, kernel_size=2, stride=2)
            )
            self.ups.append(DoubleConv3D(feature * 2, feature))

        # Фінальний шар (зводить канали до кількості наших класів)
        self.final_conv = nn.Conv3d(features[0], out_channels, kernel_size=1)

    def forward(self, x):
        skip_connections = []

        # Прохід вниз
        for down in self.downs:
            x = down(x)
            skip_connections.append(x)
            x = self.pool(x)

        # Прохід через ботлнек
        x = self.bottleneck(x)
        
        # Перевертаємо skip connections для зручності
        skip_connections = skip_connections[::-1]

        # Прохід вгору
        for i in range(0, len(self.ups), 2):
            x = self.ups[i](x)
            skip_connection = skip_connections[i//2]
            
            # Об'єднуємо ознаки з енкодера та декодера (ті самі U-подібні зв'язки)
            concat_skip = torch.cat((skip_connection, x), dim=1)
            x = self.ups[i+1](concat_skip)

        return self.final_conv(x)

# --- Блок тестування архітектури ---
if __name__ == "__main__":
    print("Ініціалізація 3D U-Net моделі...")
    model = UNet3D(in_channels=4, out_channels=4)
    
    # Створюємо фейковий тензор, який імітує батч з 1 нашого патчу
    # Розмірність: [Batch_size, Channels, Depth, Height, Width]
    dummy_input = torch.randn(1, 4, 64, 64, 64)
    
    print(f"Подаємо на вхід тензор: {dummy_input.shape}")
    
    # Пропускаємо через нейромережу
```
